# 1. Algorytm sprawdzaj¹cy czy liczba jest pierwsza
# liczba pierwsza dzieli sie tylko przez 1 i sam¹ siebie
# 2, 3, 5, 7, 11, 13, 17, 19 itd...
# liczba x ma swój dzielnik (o ile go ma) w przedziale [2, sqrt(x)]
n = int(input())
for i in range(2, n):
    if n % i == 0:
        print("liczba nie jest pierwsza")
        exit()
print("liczba jest pierwsza")

# 2. Generator liczb pierwszych - liczby pierwsze w przedziale [p, q]

p, g=map(int,input().split())
for i in range(p, g+1):
    flaga = True
    for j in range(2, int(i**0.5)+1):
        if i % j == 0:
            flaga = False
            break
    if flaga:
        print(i, end=" ")


# 3. Generator liczb pierwszych - pocz¹tkowe k liczb pierwszych

n = int(input())
x=1
while n > 0:
    flaga = True
    for i in range(2,x):
        if x % i == 0:
            flaga = False
            break
    if flaga:
        print(x,end=" ")
        n = n - 1
    x = x + 1
